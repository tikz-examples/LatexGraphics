\begin{tikzpicture}

% rectangle
\fill[lightgray] (2,0) rectangle ++(2,12);

% Right side illustration

\begin{scope}[xshift=7cm]

\draw [line width=1mm,blue] (-0.4,0) -- ++ (3.25,6.5);

% rectangle
\fill[lightgray,opacity=0.5] (2,0) rectangle ++(2,12);
\end{scope}
